Run Picoin locally or on a droplet

git clone https://github.com/devcoffeecoin/PICOIN.git
cd PICOIN/picoin-proof-of-pi
python3 -m venv .venv
source .venv/bin/activate
pip install -r requirements.txt
python -m picoin node start --host 0.0.0.0 --port 8000
python -m picoin testnet bootstrap
python -m picoin testnet continuous --miners 1 --loops 20 --faucet 0
python -m picoin node audit
